#variable declaration
name_length = 4
name = "hello"
name, name_length  = "hello", 4


#print
print(name_length)
print(name)
print(type(name_length))
print(type(name))
print('\n')


#type cast
name_length = int("4")
print(name_length)
print(type(name_length))
print('\n')


#list
name_list = ["first", "second", "third"]
print(name_list)
print(type(name_list))
print('\n')


#variable = assigning value
name1, name2, name3 = name_list
print(name1)
print(name2)
print(name3)
print('\n')


#tuple  
name_tuple = ("First", "Second", "Third")
print(name_tuple)
print(type(name_tuple))
print('\n')


#dictionary
name_dictionary = {"Python": 6, "VM": 2}
print(name_dictionary)
print(type(name_dictionary))
print('\n')


#boolean
name_boolean = True
print(name_boolean)
print(type(name_boolean))
print('\n')


#range
name_range = range(6)
print(name_range)
print(type(name_range))
print('\n')


#to be bytes, including b before the string is necessary
name_bytes = b"Kali"
print(name_bytes)
print(type(name_bytes))
